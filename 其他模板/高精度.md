class GNUM
{
private:
	static const int BASE = 10;
	static const int BASE_LEN = 1;
	static const int LEN = 20100;
	int a[LEN];
	int len;

public:
	GNUM()
	{
		len = 0;
		memset(a, 0, sizeof(a));
	}
	~GNUM()
	{
		len = 0;
		memset(a, 0, sizeof(a));
	}
	template <typename T>
	GNUM(T x)
	{
		len = 0;
		memset(a, 0, sizeof(a));
		if (x == 0)
		{
			a[++len] = x;
			return;
		}
		while (x)
		{
			a[++len] = x % BASE;
			x /= BASE;
		}
	}
	template <typename Y>
	GNUM(Y *s, int lenth)
	{
		len = lenth;
		memcpy(a, s, sizeof(a));
	}
	GNUM(char *x)
	{
		len = 1;
		register int lenth = strlen(x) - 1;
		memset(a, 0, sizeof(a));
		for (register int i = lenth, j = 1; i >= 0; --i)
		{
			if (j == BASE)
				j = 1, ++len;
			a[len] += (x[i] - '0') * j;
			j *= 10;
		}
	}
	friend istream &operator>>(istream &input, GNUM &in)

		char s[LEN];
		input >> s;
		in = GNUM(s);
		return input;
	}

	}

	}

		while (!a[len] && len > 1)
			--len;
	}

		if (!k && len == 1 && !a[1])
			return;
		for (register int i = len; i; --i)
			a[i + 1] = a[i];
		a[1] = k;
		++len;
	}

		}

		return res;
	}

		}

		return res;

		{
			k.add(x.a[i]);
			while (k >= y)
				k = k - y, res.a[i]++;
		}
		res.del_zero();
		return res;
	}
